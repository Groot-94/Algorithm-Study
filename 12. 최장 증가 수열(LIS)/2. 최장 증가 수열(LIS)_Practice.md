### 예톤

- LIS(최장 증가 수열) + DP(동적 계획법) 사용
```swift
func LIS(arr: [Int]) -> Int {
    var dp = [Int](repeating: 1, count: arr.count)
    
    for i in 1..<arr.count {
        for j in 0..<i {
            if arr[i] > arr[j] {
                dp[i] = max(dp[i], dp[j]+1)
            }
        }
    }
    
    return dp.max()!
}
```

### 수꿍

- 구현(DP(동적 계획법) 사용)
```swift
let array = [3, 5, 7, 9, 2, 1, 4, 8]
var dp = [Int](repeating: 1, count: array.count)

for i in 1..<array.count {
    for j in 0..<i {
        if array[i] > array[j] {
            dp[i] = dp[i] > dp[j] + 1 ? dp[i] : dp[j] + 1
        }
    }
}

print(dp.max()!)
```

- 설명
```swift
let array = [3, 5, 7, 9, 2, 1, 4, 8]
// LIS 결과를 담기 위한 배열 생성
var dp = [Int](repeating: 1, count: array.count)

// 배열의 임의의 요소를 택하여
// 그 요소 전까지의 부분증가수열 갯수를 저장
// i가 더 큰 범위
// j가 더 작은 범위
// 0번 인덱스는 어차피 의미가 없어
// 1번 인덱스부터 시작되도록 설정
for i in 1..<array.count {
    for j in 0..<i {
        // 증가 수열일 경우만을 위한 조건문
        if array[i] > array[j] {
            // 이전까지의 LIS 결과값을 비교하여
            // 더 큰 값을 저장
            dp[i] = dp[i] > dp[j] + 1 ? dp[i] : dp[j] + 1
        }
    }
}

print(dp.max()!)
```
