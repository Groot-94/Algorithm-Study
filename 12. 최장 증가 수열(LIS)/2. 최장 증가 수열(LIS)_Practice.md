### 예톤

- LIS(최장 증가 수열) + DP(동적 계획법) 사용
```swift
func LIS(arr: [Int]) -> Int {
    var dp = [Int](repeating: 1, count: arr.count)
    
    for i in 1..<arr.count {
        for j in 0..<i {
            if arr[i] > arr[j] {
                dp[i] = max(dp[i], dp[j]+1)
            }
        }
    }
    
    return dp.max()!
}
```
